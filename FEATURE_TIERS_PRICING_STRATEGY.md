// lib/featureGating.ts
export async function hasFeature(
  tenantId: string,
  feature: 'pdfReports' | 'emailNotifications' | 'skills' | 'attendance' | 'analytics' | 'feeManagement'
): Promise<boolean> {
  // Check tenant's subscription
  const subscription = await getSubscription(tenantId);

  // Check if feature is included in plan or purchased as add-on
  return subscription.plan.includes(feature) || subscription.addOns.includes(feature);
}

// Usage in components

  // Show PDF download button
